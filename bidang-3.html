Sintaks R

Visualisasi indikator manusia

pembangunan_manusia.R
library(ggplot2)

manusia <- read.csv("contoh_pembangunan_manusia.csv")

ggplot(manusia, aes(reorder(indikator, nilai), nilai)) +
  geom_col(fill = "#9b4676", width = .65) +
  coord_flip() +
  geom_text(aes(label = nilai), hjust = -0.2) +
  labs(x = NULL, y = "Nilai indikator") + theme_minimal()
